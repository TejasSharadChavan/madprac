import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'DartPad UI Demo',
      theme: ThemeData(
        primarySwatch: Colors.teal,
      ),
      home: HomePage(),
      debugShowCheckedModeBanner: false,
    );
  }
}

class HomePage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Flutter UI Example'),
        actions: [
          IconButton(
            icon: Icon(Icons.settings),
            onPressed: () {},
          ),
        ],
      ),
      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            // Profile image and name
            Row(
              children: [
                CircleAvatar(
                  radius: 40,
                  backgroundImage:
                      NetworkImage('https://i.pravatar.cc/150?img=3'),
                ),
                SizedBox(width: 16),
                Text(
                  'John Doe',
                  style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold),
                ),
              ],
            ),

            SizedBox(height: 24),

            // Icons row
            Row(
              mainAxisAlignment: MainAxisAlignment.spaceEvenly,
              children: [
                Column(children: [
                  Icon(Icons.email, color: Colors.teal),
                  Text('Email'),
                ]),
                Column(children: [
                  Icon(Icons.phone, color: Colors.teal),
                  Text('Call'),
                ]),
                Column(children: [
                  Icon(Icons.share, color: Colors.teal),
                  Text('Share'),
                ]),
              ],
            ),

            SizedBox(height: 40),

            // Simulated Chart
            Text('Activity Chart', style: TextStyle(fontSize: 18)),
            SizedBox(height: 16),

            _buildChartRow('Work', Colors.blue, 0.9),
            _buildChartRow('Study', Colors.green, 0.7),
            _buildChartRow('Exercise', Colors.orange, 0.5),
            _buildChartRow('Leisure', Colors.red, 0.3),
          ],
        ),
      ),
    );
  }

  // Helper method for reusable chart rows
  Widget _buildChartRow(String label, Color color, double fillPercent) {
    return Padding(
      padding: const EdgeInsets.only(bottom: 8),
      child: Row(
        children: [
          SizedBox(width: 70, child: Text(label)),
          Expanded(
            child: FractionallySizedBox(
              alignment: Alignment.centerLeft,
              widthFactor: fillPercent,
              child: Container(
                height: 20,
                color: color,
              ),
            ),
          ),
        ],
      ),
    );
  }
}
